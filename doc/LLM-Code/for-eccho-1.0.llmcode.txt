import eccho

# Create an LLM-Code interpreter
interpreter = eccho.LLMCodeInterpreter()

# Load a LLM-Code script
script = """
# Useless comment
import numpy as np

# Create a tensor
x = np.array([1, 2, 3])

# Print the tensor
print(x)
"""

# Interpret the script
result = interpreter.interpret(script)

# Print the result
print(result)
